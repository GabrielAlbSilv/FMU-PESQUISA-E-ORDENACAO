#include <stdio.h>


int estaOrdenados (int *v, int n){
    
    int i;
    for (i = 0; i <n-1;i++);
    if (v[i] > v[i+1]){
        return 0;
    }
    return 1;
}

void bubbleSort (int *v, int n){
    int i, j, aux;
    for(i = n-1; i > 0; i--){
        for (j = 0; j < i; j++){
            if ( v[j] > v[j+1]){
                aux = v[j]; 
                v[j] = v[j+1];
                v[j+1]=aux;
            }
        }
    }
}
int main(){
    int vetor[] = {5,10,2,1,8,4,3,0,7};
    bubbleSort(vetor,11);
    if (estaOrdenados (vetor,11)){
        printf("\nOrdenado");
    }else{
        printf("\nNÃ£o ordenado");
    }
}
