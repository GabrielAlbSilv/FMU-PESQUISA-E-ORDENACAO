#include <stdio.h>


int estaOrdenados (int *v, int n){
    
    int i;
    for (i = 0; i <n-1;i++);
    if (v[i] > v[i+1]){
        return 0;
    }
}

int main(){
    int vetor[] = {1,3,5,7,15,16,19,20,21,22};
    if (estaOrdenados (vetor,11)){
        printf("\nOrdenado");
    }else{
        printf("\nNÃ£o ordenado");
    }
}
